# Examples

## Using a saga

```javascript
import { connectTrixta } from '@trixtateam/trixta-js';
import { put } from 'redux-saga/effects';
yield put(connectTrixta({space:'trixta-demo.space.trixta.io', params:{} }));

```

## Using useDispatch hook

```javascript
import { connectTrixta } from '@trixtateam/trixta-js';
import { useDispatch } from 'react-redux';

const dispatch = useDispatch();
dispatch(connectTrixta({space:'trixta-demo.space.trixta.io', params:{} }))

```
