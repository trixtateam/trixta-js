# Examples

## Using a saga

```javascript
import { updateTrixtaRole } from '@trixtateam/trixta-js';
import { put } from 'redux-saga/effects';
const roles = [{name:'name of Trixta role 1'},{name:'name of Trixta role 2'}];
yield put(updateTrixtaRoles({ roles }));

```

## Using useDispatch hook

```javascript
import { updateTrixtaRoles } from '@trixtateam/trixta-js';
import { useDispatch } from 'react-redux';

const dispatch = useDispatch();
const roles = [
  { name: 'name of Trixta role 1' },
  { name: 'name of Trixta role 2' },
];
dispatch(updateTrixtaRoles({ roles }));
```
