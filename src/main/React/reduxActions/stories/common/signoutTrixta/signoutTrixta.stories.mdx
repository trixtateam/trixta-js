# Examples

## Using a saga

```javascript
import { signoutTrixta } from '@trixtateam/trixta-js';
import { put } from 'redux-saga/effects';
yield put(signoutTrixta())

```

## Using useDispatch hook

```javascript
import { signoutTrixta } from '@trixtateam/trixta-js';
import { useDispatch } from 'react-redux';

const dispatch = useDispatch();
dispatch(signoutTrixta())
```
