# Examples

## Using a saga

```javascript
import { updateTrixtaRole } from '@trixtateam/trixta-js';
import { put } from 'redux-saga/effects';

yield put(updateTrixtaRole({name:'name of Trixta role'}))

```

## Using useDispatch hook

```javascript
import { updateTrixtaRole } from '@trixtateam/trixta-js';
import { useDispatch } from 'react-redux';
const dispatch = useDispatch();

dispatch(updateTrixtaRole({ name: 'name of Trixta role' }));
```
